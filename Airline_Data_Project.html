Calculate the average occupancy per aircraft.

In[52]:
occupancy_rate=pd.read_sql("""with cte as (
              select aircraft_code,f.flight_id, count(*) as seat_count from boarding_passes as b join flights as f on b.flight_id=f.flight_id 
              group by aircraft_code,f.flight_id), cte1 as 
              (
               select aircraft_code, count(*) as num_seats from seats group by aircraft_code)
               select c1.aircraft_code,avg(c1.seat_count)as booked_seat,c2.num_seats, 
               avg(c1.seat_count)/c2.num_seats as occupancy_rate from cte as c1
               join cte1 as c2
               on c1.aircraft_code=c2.aircraft_code group by c1.aircraft_code,c2.num_seats""",connection)
